    function pad(n) { return String(n).padStart(2, "0"); }


    function AnalogeUhr({ stunde, minute, onMinuteDrag, onStundeDrag, interactive, dragHint, show24 }) {

      const svgRef = useRef(null);

      const dragging = useRef(null);

      const lastAngle = useRef(null);

      const cx = 120, cy = 120, r = 100;

      const minAngle = (minute / 60) * 360 - 90;

      const stdAngle = ((stunde % 12 + minute / 60) / 12) * 360 - 90;

      const minRad = (minAngle * Math.PI) / 180;

      const stdRad = (stdAngle * Math.PI) / 180;


      const getSVGAngle = (e) => {

        const rect = svgRef.current.getBoundingClientRect();

        const scale = 240 / rect.width;

        const clientX = e.touches ? e.touches[0].clientX : e.clientX;

        const clientY = e.touches ? e.touches[0].clientY : e.clientY;

        return Math.atan2((clientY - rect.top) * scale - cy, (clientX - rect.left) * scale - cx);

      };


      const onPointerDown = (type) => (e) => {

        if (!interactive) return;

        dragging.current = type;

        lastAngle.current = getSVGAngle(e);

        e.preventDefault();

      };


      const onPointerMove = useCallback((e) => {

        if (!dragging.current || !svgRef.current) return;

        const angle = getSVGAngle(e);

        let delta = angle - lastAngle.current;

        if (delta > Math.PI) delta -= 2 * Math.PI;

        if (delta < -Math.PI) delta += 2 * Math.PI;

        lastAngle.current = angle;

        const deg = (delta * 180) / Math.PI;

        if (dragging.current === "minute") onMinuteDrag(deg);

        else onStundeDrag(deg);

      }, [onMinuteDrag, onStundeDrag]);


      const onPointerUp = useCallback(() => { dragging.current = null; lastAngle.current = null; }, []);


      useEffect(() => {

        window.addEventListener("mousemove", onPointerMove);

        window.addEventListener("mouseup", onPointerUp);

        window.addEventListener("touchmove", onPointerMove, { passive: false });

        window.addEventListener("touchend", onPointerUp);

        return () => {

          window.removeEventListener("mousemove", onPointerMove);

          window.removeEventListener("mouseup", onPointerUp);

          window.removeEventListener("touchmove", onPointerMove);

          window.removeEventListener("touchend", onPointerUp);

        };

      }, [onPointerMove, onPointerUp]);


      const zahlen = [];

      for (let i = 1; i <= 12; i++) {

        const a = ((i / 12) * 2 * Math.PI) - Math.PI / 2;

        zahlen.push({ i, x: cx + 80 * Math.cos(a), y: cy + 80 * Math.sin(a) });

      }

      const zahlen24 = [];

      if (show24) {

        for (let i = 1; i <= 12; i++) {

          const a = ((i / 12) * 2 * Math.PI) - Math.PI / 2;

          const v = i + 12 === 24 ? 0 : i + 12;

          zahlen24.push({ i: v, x: cx + 62 * Math.cos(a), y: cy + 62 * Math.sin(a) });

        }

      }

      const striche = [];

      for (let i = 0; i < 60; i++) {

        const a = (i / 60) * 2 * Math.PI - Math.PI / 2;

        const inner = i % 5 === 0 ? 88 : 94;

        striche.push({ x1: cx + inner * Math.cos(a), y1: cy + inner * Math.sin(a), x2: cx + r * Math.cos(a), y2: cy + r * Math.sin(a), bold: i % 5 === 0 });

      }


      return (

        <div style={{ textAlign: "center", display: "flex", flexDirection: "column", alignItems: "center" }}>

          <svg ref={svgRef} viewBox="0 0 240 240" style={{ width: "100%", maxWidth: 260, cursor: interactive ? "pointer" : "default", userSelect: "none" }}>

            <circle cx={cx} cy={cy} r={r} fill="#fff9f0" stroke="#e8a825" strokeWidth="3" />

            {striche.map((s, i) => (

              <line key={i} x1={s.x1} y1={s.y1} x2={s.x2} y2={s.y2} stroke="#c97a10" strokeWidth={s.bold ? 2 : 1} strokeLinecap="round" />

            ))}

            {zahlen.map(z => (

              <text key={z.i} x={z.x} y={z.y} textAnchor="middle" dominantBaseline="central" fontSize="14" fontWeight="500" fill="#5a3e0a">{z.i}</text>

            ))}

            {zahlen24.map(z => (

              <text key={"24-" + z.i} x={z.x} y={z.y} textAnchor="middle" dominantBaseline="central" fontSize="9" fontWeight="400" fill="#c97a10">{z.i}</text>

            ))}

            <line x1={cx} y1={cy} x2={cx + 55 * Math.cos(stdRad)} y2={cy + 55 * Math.sin(stdRad)} stroke="#3d2a08" strokeWidth="6" strokeLinecap="round" onMouseDown={onPointerDown("stunde")} onTouchStart={onPointerDown("stunde")} />

            <line x1={cx} y1={cy} x2={cx + 75 * Math.cos(minRad)} y2={cy + 75 * Math.sin(minRad)} stroke="#c97a10" strokeWidth="4" strokeLinecap="round" onMouseDown={onPointerDown("minute")} onTouchStart={onPointerDown("minute")} />

            <circle cx={cx} cy={cy} r="6" fill="#5a3e0a" />

          </svg>

          {interactive && (

            <p style={{ fontSize: 11, color: "#c97a10", margin: "2px 0 0", textAlign: "center" }}>{dragHint}</p>

          )}

        </div>

      );

    }


    function DigitaleUhr({ stunde, minute }) {

      return (

        <div style={{ background: "#1a1a2e", color: "#ffd700", borderRadius: 12, padding: "10px 28px", fontSize: 38, fontWeight: 500, fontFamily: "monospace", letterSpacing: 4, textAlign: "center", display: "inline-block" }}>

          {pad(stunde)}:{pad(minute)}

        </div>

      );

    }


    function zufaelligeUhrzeit() {

      return { h: Math.floor(Math.random() * 12) + 1, m: Math.floor(Math.random() * 12) * 5 };

    }


    function ErkundenModus({ lang }) {

      const t = T[lang];

      const [stunde, setStunde] = useState(3);

      const [minute, setMinute] = useState(0);

      const minuteAcc = useRef(0);

      const stundeAcc = useRef(0);

      const isNachmittag = stunde >= 12;


      const handleMinuteDrag = useCallback((deg) => {

        minuteAcc.current += deg;

        const steps = Math.trunc(minuteAcc.current / 6);

        if (steps !== 0) {

          minuteAcc.current -= steps * 6;

          setMinute(m => {

            let nm = m + steps;

            if (nm >= 60) { nm = nm % 60; setStunde(h => Math.min(23, h + 1)); }

            if (nm < 0) { nm = ((nm % 60) + 60) % 60; setStunde(h => Math.max(0, h - 1)); }

            return nm;

          });

        }

      }, []);


      const handleStundeDrag = useCallback((deg) => {

        stundeAcc.current += deg;

        const steps = Math.trunc(stundeAcc.current / 30);

        if (steps !== 0) {

          stundeAcc.current -= steps * 30;

          setStunde(h => Math.min(23, Math.max(0, h + steps)));

        }

      }, []);


      const toggle = () => setStunde(h => h >= 12 ? h - 12 : h + 12);


      return (

        <div style={{ textAlign: "center" }}>

          <p style={{ color: "#888", fontSize: 14, margin: "0 0 8px", textAlign: "center" }}>{t.exploreHint}</p>

          <div style={{ display: "flex", justifyContent: "center", gap: 8, marginBottom: 10 }}>

            {[false, true].map((pm, i) => (

              <button key={i} onClick={() => (pm !== isNachmittag) && toggle()}

                style={{ padding: "5px 14px", borderRadius: 20, border: "2px solid #e8a825", fontSize: 12, fontWeight: 500, cursor: "pointer", background: isNachmittag === pm ? "#f5a623" : "#fff9f0", color: isNachmittag === pm ? "#fff" : "#7a5a20" }}>

                {pm ? t.nachmittag : t.vormittag}

              </button>

            ))}

          </div>

          <AnalogeUhr stunde={stunde} minute={minute} onMinuteDrag={handleMinuteDrag} onStundeDrag={handleStundeDrag} interactive={true} dragHint={t.dragHint} show24={lang === "de"} />

          <div style={{ margin: "10px 0 8px" }}><DigitaleUhr stunde={stunde} minute={minute} /></div>

          <div style={{ margin: "6px auto 0", background: "#fef3d0", borderRadius: 10, padding: "8px 20px", display: "inline-block", fontSize: 18, color: "#7a4f00", fontWeight: 500 }}>

            {t.timeText(stunde, minute)} · {isNachmittag ? t.nachLabel : t.vormLabel}

          </div>

        </div>

      );

    }


    function AblesenModus({ lang }) {

      const t = T[lang];

      const [uhr, setUhr] = useState(zufaelligeUhrzeit);

      const [eingabe, setEingabe] = useState("");

      const [status, setStatus] = useState(null);

      const [punkte, setPunkte] = useState(0);

      const [runde, setRunde] = useState(1);


      const pruefen = () => {

        const v = eingabe.trim().replace(/\s+/g, "");

        const r1 = pad(uhr.h) + ":" + pad(uhr.m);

        const r2 = uhr.h + ":" + pad(uhr.m);

        const ok = v === r1 || v === r2 || v === r1.replace(":", "") || v === (uhr.h + pad(uhr.m));

        setStatus(ok ? "richtig" : "falsch");

        if (ok) setPunkte(p => p + 1);

      };


      const naechste = () => { setUhr(zufaelligeUhrzeit()); setEingabe(""); setStatus(null); setRunde(r => r + 1); };


      return (

        <div style={{ textAlign: "center" }}>

          <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 8, fontSize: 13, color: "#888" }}>

            <span>{t.round} {runde}</span><span>{t.points}: {punkte} ⭐</span>

          </div>

          <p style={{ color: "#888", fontSize: 14, margin: "0 0 8px", textAlign: "center" }}>{t.readHint}</p>

          <AnalogeUhr stunde={uhr.h} minute={uhr.m} onMinuteDrag={() => {}} onStundeDrag={() => {}} interactive={false} dragHint="" show24={false} />

          {status === null ? (

            <div style={{ display: "flex", gap: 8, justifyContent: "center", margin: "10px 0" }}>

              <input value={eingabe} onChange={e => setEingabe(e.target.value)} onKeyDown={e => e.key === "Enter" && eingabe && pruefen()} placeholder={t.placeholder}

                style={{ fontSize: 20, padding: "6px 14px", borderRadius: 10, border: "2px solid #e8a825", width: 110, textAlign: "center" }} />

              <button onClick={pruefen} disabled={!eingabe} style={{ background: "#f5a623", color: "#fff", border: "none", borderRadius: 10, padding: "6px 18px", fontSize: 18, cursor: "pointer", fontWeight: 500 }}>✓</button>

            </div>

          ) : (

            <div style={{ margin: "10px 0" }}>

              <div style={{ background: status === "richtig" ? "#e8f5e9" : "#fdecea", color: status === "richtig" ? "#2e7d32" : "#c62828", borderRadius: 10, padding: "10px 20px", fontSize: 18, fontWeight: 500, marginBottom: 10 }}>

                {status === "richtig" ? t.correct : t.wrong(pad(uhr.h) + ":" + pad(uhr.m))}

              </div>

              <button onClick={naechste} style={{ background: "#f5a623", color: "#fff", border: "none", borderRadius: 10, padding: "8px 24px", fontSize: 16, cursor: "pointer", fontWeight: 500 }}>{t.next}</button>

            </div>

          )}

        </div>

      );

    }


    function QuizModus({ lang }) {

      const t = T[lang];


      const genFrage = () => {

        const richtig = zufaelligeUhrzeit();

        const falschen = [];

        while (falschen.length < 3) {

          const f = zufaelligeUhrzeit();

          if (!falschen.find(x => x.h === f.h && x.m === f.m) && !(f.h === richtig.h && f.m === richtig.m)) falschen.push(f);

        }

        return { richtig, optionen: [richtig, ...falschen].sort(() => Math.random() - 0.5) };

      };


      const [frage, setFrage] = useState(genFrage);

      const [gewaehlt, setGewaehlt] = useState(null);

      const [punkte, setPunkte] = useState(0);

      const [runde, setRunde] = useState(1);

      const [sterne, setSterne] = useState(0);


      const waehlen = (opt) => {

        if (gewaehlt) return;

        setGewaehlt(opt);

        const ok = opt.h === frage.richtig.h && opt.m === frage.richtig.m;

        if (ok) { setPunkte(p => p + 1); setSterne(s => Math.min(3, s + 1)); } else setSterne(s => Math.max(0, s - 1));

      };


      const naechste = () => { setFrage(genFrage()); setGewaehlt(null); setRunde(r => r + 1); };

      const isRight = gewaehlt && gewaehlt.h === frage.richtig.h && gewaehlt.m === frage.richtig.m;

      const richtigStr = pad(frage.richtig.h) + ":" + pad(frage.richtig.m);


      return (

        <div style={{ textAlign: "center" }}>

          <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 8, fontSize: 13, color: "#888" }}>

            <span>{t.round} {runde}</span>

            <span>

              {Array.from({ length: 3 }).map((_, i) => (

                <span key={i} style={{ color: i < sterne ? "#f5a623" : "#ddd" }}>★</span>

              ))}

              {" "}{punkte} {t.points}

            </span>

          </div>

          {runde % 2 === 0 ? (

            <div>

              <p style={{ color: "#888", fontSize: 14, margin: "0 0 6px" }}>{t.quizQ1}</p>

              <AnalogeUhr stunde={frage.richtig.h} minute={frage.richtig.m} onMinuteDrag={() => {}} onStundeDrag={() => {}} interactive={false} dragHint="" show24={false} />

              <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 10, marginTop: 10 }}>

                {frage.optionen.map((opt, i) => {

                  const ir = opt.h === frage.richtig.h && opt.m === frage.richtig.m;

                  const gd = gewaehlt && gewaehlt.h === opt.h && gewaehlt.m === opt.m;

                  let bg = "#fff9f0", border = "2px solid #e8a825", color = "#5a3e0a";

                  if (gewaehlt) {

                    if (ir) { bg = "#e8f5e9"; border = "2px solid #4caf50"; color = "#2e7d32"; }

                    else if (gd) { bg = "#fdecea"; border = "2px solid #f44336"; color = "#c62828"; }

                    else { bg = "#f5f5f5"; color = "#aaa"; border = "2px solid #ddd"; }

                  }

                  return (

                    <button key={i} onClick={() => waehlen(opt)} style={{ background: bg, border, color, borderRadius: 12, padding: "12px 0", fontSize: 18, fontWeight: 500, cursor: gewaehlt ? "default" : "pointer", fontFamily: "monospace" }}>

                      {pad(opt.h)}:{pad(opt.m)}

                    </button>

                  );

                })}

              </div>

            </div>

          ) : (

            <div>

              <p style={{ color: "#888", fontSize: 14, margin: "0 0 8px" }}>

                {t.quizQ2pre}{" "}<span style={{ color: "#c97a10", fontWeight: 500, fontSize: 18 }}>{richtigStr}</span>?

              </p>

              <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 8 }}>

                {frage.optionen.map((opt, i) => {

                  const ir = opt.h === frage.richtig.h && opt.m === frage.richtig.m;

                  const gd = gewaehlt && gewaehlt.h === opt.h && gewaehlt.m === opt.m;

                  let border = "2px solid #e8a825", bg = "#fff9f0";

                  if (gewaehlt) {

                    if (ir) { bg = "#e8f5e9"; border = "2px solid #4caf50"; }

                    else if (gd) { bg = "#fdecea"; border = "2px solid #f44336"; }

                    else { bg = "#f5f5f5"; border = "2px solid #ddd"; }

                  }

                  return (

                    <div key={i} onClick={() => waehlen(opt)} style={{ background: bg, border, borderRadius: 12, padding: 6, cursor: gewaehlt ? "default" : "pointer", display: "flex", justifyContent: "center" }}>

                      <AnalogeUhr stunde={opt.h} minute={opt.m} onMinuteDrag={() => {}} onStundeDrag={() => {}} interactive={false} dragHint="" show24={false} />

                    </div>

                  );

                })}

              </div>

            </div>

          )}

          {gewaehlt && (

            <div style={{ marginTop: 10 }}>

              <div style={{ background: isRight ? "#e8f5e9" : "#fdecea", color: isRight ? "#2e7d32" : "#c62828", borderRadius: 10, padding: "8px 20px", fontSize: 17, fontWeight: 500, marginBottom: 10 }}>
